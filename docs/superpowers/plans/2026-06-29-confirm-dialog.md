# Confirm Dialog Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace Turbo's native `window.confirm` with a DaisyUI-styled `<dialog>` shipped by Bali, so all `data-turbo-confirm` confirmations look consistent and are operable by automated browser tools.

**Architecture:** A small JS module assigns a custom async function to `Turbo.config.forms.confirm`. The function renders a singleton native `<dialog class="modal">` (top-layer, real DOM) and resolves a `Promise<boolean>`. The install is triggered automatically from Bali's `registerAll` so consuming apps need no changes. `DeleteLink` is updated to emit customization data-attributes (red destructive button + localized title/labels). A Lookbook preview demos it for delete and non-delete actions.

**Tech Stack:** Ruby on Rails (ViewComponent engine), Stimulus/Turbo 8 (`@hotwired/turbo-rails ^8.0.21`), DaisyUI + Tailwind, Minitest, Cypress, Lookbook.

## Global Constraints

- JS style: `standard` linter — **no semicolons**, single quotes, space before function parens. Match `app/components/bali/modal/index.js`.
- Turbo 8 confirm API (verified in `spec/dummy/node_modules/@hotwired/turbo/dist/turbo.es2017-esm.js:1157-1161`): assign `Turbo.config.forms.confirm`; signature `(message, formElement, submitter)`; result is `await`ed → `Promise<boolean>` supported. `Turbo.setConfirmMethod` is deprecated.
- Data-attribute contract (namespaced `data-bali-confirm-*`): `title`, `variant` (`danger`→`btn-error` / `warning`→`btn-warning` / `info`→`btn-info`, default `btn-primary`), `accept`, `cancel`. Resolution order: `submitter` → `formElement` → install-time default.
- Opt-out: `window.BALI_DISABLE_CONFIRM_DIALOG` truthy disables install.
- DaisyUI component classes (`modal`, `modal-box`, `modal-action`, `modal-backdrop`, `btn`, `btn-error`…) are generated by the daisyUI plugin and always available. Only pure Tailwind utilities used in JS template strings depend on content scanning.
- After CSS-affecting changes: `bundle exec rails tailwindcss:build`.
- Browser verification is MANDATORY (per `.claude/CLAUDE.md`) before claiming done.
- Out of scope: `Bali::TurboNativeApp::SignOut` (`app/components/bali/turbo_native_app/sign_out/index.js:10`) keeps its own `window.confirm` — Turbo Native path, not changed here.

```js
/**
 * Bali custom confirmation dialog
 *
 * Replaces Turbo's default window.confirm with a DaisyUI-styled native <dialog>.
 * It renders as real DOM in the top layer, so automated browser tools (e.g. Claude
 * in Chrome) can operate it — unlike the native window.confirm modal.
 *
 * Wired into Turbo via: Turbo.config.forms.confirm = confirmDialog
 *
 * Per-trigger customization via data-attributes on the submitter or form:
 *   data-bali-confirm-title, data-bali-confirm-variant (danger|warning|info),
 *   data-bali-confirm-accept, data-bali-confirm-cancel
 */

const VARIANT_CLASSES = {
  danger: 'btn-error',
  warning: 'btn-warning',
  info: 'btn-info'
}

const DEFAULT_ACCEPT_CLASS = 'btn-primary'

const labels = {
  accept: 'OK',
  cancel: 'Cancel'
}

let dialog = null
let titleEl = null
let messageEl = null
let acceptBtn = null
let cancelBtn = null
let activeResolve = null
let confirmed = false
let installed = false

function buildDialog () {
  dialog = document.createElement('dialog')
  dialog.className = 'modal'
  dialog.dataset.baliConfirm = ''
  dialog.innerHTML = `
    <div class="modal-box" role="alertdialog" aria-labelledby="bali-confirm-title" aria-describedby="bali-confirm-message">
      <h3 id="bali-confirm-title" class="text-lg font-bold"></h3>
      <p id="bali-confirm-message" class="py-4 whitespace-pre-line"></p>
      <div class="modal-action">
        <button type="button" class="btn btn-ghost" data-bali-confirm-cancel></button>
        <button type="button" class="btn" data-bali-confirm-accept></button>
      </div>
    </div>
    <form method="dialog" class="modal-backdrop">
      <button tabindex="-1" aria-hidden="true">close</button>
    </form>
  `

  titleEl = dialog.querySelector('#bali-confirm-title')
  messageEl = dialog.querySelector('#bali-confirm-message')
  cancelBtn = dialog.querySelector('[data-bali-confirm-cancel]')
  acceptBtn = dialog.querySelector('[data-bali-confirm-accept]')

  acceptBtn.addEventListener('click', () => {
    confirmed = true
    dialog.close()
  })

  cancelBtn.addEventListener('click', () => {
    confirmed = false
    dialog.close()
  })

  // The close event fires for button clicks, ESC, and backdrop submit.
  dialog.addEventListener('close', () => {
    const resolve = activeResolve
    activeResolve = null
    if (resolve) resolve(confirmed)
  })

  document.body.appendChild(dialog)
}

function dataValue (submitter, formElement, key) {
  if (submitter && submitter.dataset[key]) return submitter.dataset[key]
  if (formElement && formElement.dataset[key]) return formElement.dataset[key]
  return null
}

export function confirmDialog (message, formElement, submitter) {
  if (!dialog) buildDialog()

  const title = dataValue(submitter, formElement, 'baliConfirmTitle')
  const variant = dataValue(submitter, formElement, 'baliConfirmVariant')

  titleEl.textContent = title || ''
  titleEl.hidden = !title

  messageEl.textContent = message || ''

  acceptBtn.textContent = dataValue(submitter, formElement, 'baliConfirmAccept') || labels.accept
  cancelBtn.textContent = dataValue(submitter, formElement, 'baliConfirmCancel') || labels.cancel

  acceptBtn.className = 'btn'
  acceptBtn.classList.add(VARIANT_CLASSES[variant] || DEFAULT_ACCEPT_CLASS)

  confirmed = false

  return new Promise(resolve => {
    activeResolve = resolve
    dialog.showModal()
    cancelBtn.focus()
  })
}

export function installConfirmDialog (options = {}) {
  if (options.acceptText) labels.accept = options.acceptText
  if (options.cancelText) labels.cancel = options.cancelText

  if (typeof window === 'undefined' || window.BALI_DISABLE_CONFIRM_DIALOG) return
  if (installed) return
  installed = true

  const apply = () => {
    if (window.Turbo && window.Turbo.config && window.Turbo.config.forms) {
      window.Turbo.config.forms.confirm = confirmDialog
      return true
    }
    return false
  }

  if (!apply()) {
    const onLoad = () => {
      document.removeEventListener('turbo:load', onLoad)
      apply()
    }
    document.addEventListener('turbo:load', onLoad)
  }
}
```

- [ ] **Step 1: Write the spec**

```js
describe('Confirm dialog', () => {
  beforeEach(() => {
    cy.visit('/bali/confirm_dialog/default')
  })

  it('shows a styled dialog instead of the native confirm', () => {
    cy.contains('button', 'Eliminar registro').click()

    cy.get('dialog.modal[open]').should('be.visible')
    cy.get('#bali-confirm-title').should('have.text', 'Eliminar registro')
    cy.get('#bali-confirm-message').should('contain', 'eliminar este registro')
    cy.get('[data-bali-confirm-accept]')
      .should('have.class', 'btn-error')
      .and('have.text', 'Sí, eliminar')
  })

  it('cancelling closes the dialog', () => {
    cy.contains('button', 'Eliminar registro').click()
    cy.get('dialog.modal[open]').should('be.visible')

    cy.get('[data-bali-confirm-cancel]').click()
    cy.get('dialog.modal[open]').should('not.exist')
    // Page intact — no navigation happened
    cy.contains('button', 'Eliminar registro').should('be.visible')
  })

  it('confirming proceeds with the action', () => {
    cy.intercept('GET', '**/bali/confirm_dialog/default*').as('go')
    cy.contains('button', 'Publicar cambios').click()

    cy.get('dialog.modal[open]').should('be.visible')
    cy.get('[data-bali-confirm-accept]')
      .should('have.class', 'btn-primary')
      .click()

    cy.wait('@go')
    cy.get('dialog.modal[open]').should('not.exist')
  })
})
```
